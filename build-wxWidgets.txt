WINDOWS (Ruby Devkit)
---------------------

 Prerequisities:
 - RubyInstaller with Devkit
 - SWIG for windows (binary in path)
 - Doxygen for windows (binary in path)
 - wxWidgets source installation

 ridk exec bash -c 'pacman -S --needed --noconfirm ${MINGW_PACKAGE_PREFIX}-libjpeg-turbo'
 ridk exec bash -c 'pacman -S --needed --noconfirm ${MINGW_PACKAGE_PREFIX}-libpng'
 ridk exec bash -c 'pacman -S --needed --noconfirm ${MINGW_PACKAGE_PREFIX}-libtiff'

 cd %WXWIN%
 mkdir build-msw
 cd build-msw
 ridk exec bash -c '../configure --prefix=<installdir> --disable-tests --without-subdirs'
 ridk exec bash -c 'make'
 ridk exec bash -c 'make install'

need RUBY_DLL_PATH when requiring DLLs outside normal Ruby extension folders or system folders
